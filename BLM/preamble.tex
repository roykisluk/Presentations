% Transition slide macro
\newcommand{\transitionslide}[2]{
    \begingroup
    \setbeamercolor{background canvas}{bg=DeepNavy}
    \setbeamercolor{normal text}{fg=Cream}
    \begin{frame}[c, plain]
        \centering
        {\Huge\bfseries\color{Cream} #1}
        
        \vspace{0.5cm}
        {\Large\color{Teal} #2}
    \end{frame}
    \endgroup
}
