\section{Environment}
\begin{itemize}
    \item \textbf{Time:} Continuous.
    \item \textbf{Sectors:} $j \in \{H, L\}$ denoting High-tech and Low-tech (non-high-tech). Sector-specific productivity factors are $p_H$ and $p_L$.
    \item \textbf{Workers:} Unit mass of risk-neutral workers. There are 3 global worker types (abilities) indexed by $i \in \{1, 2, 3\}$, where ability scalars are $a_1 < a_2 < a_3$.
    \item \textbf{Labor Force:} The total mass of type $i$ workers is $m_i$. Workers discount the future at rate $r$. Unemployed workers receive a flow benefit $b$.
    \item \textbf{Firms:} Large pool of risk-neutral potential firms. A firm can open a vacancy in sector $j$ at a flow cost $c_j$.
\end{itemize}

\section{Search and Matching}
Workers direct their search to a specific sector based on expected utility, but match randomly with firms within that sector.
\begin{itemize}
    \item \textbf{Sector Choice:} Let $\phi_{iH} \in [0, 1]$ be the probability/fraction of unemployed type $i$ workers who choose to search in sector $H$, and $\phi_{iL} = 1 - \phi_{iH}$ for sector $L$.
    \item \textbf{Sector Pools:} The total mass of unemployed workers searching in sector $j$ is $u_j = \sum_{i=1}^3 \phi_{ij} u_i$, where $u_i$ is the number of unemployed type $i$ workers.
    \item \textbf{Market Tightness:} Let $v_j$ be the number of vacancies in sector $j$. Market tightness is $\theta_j = v_j / u_j$.
    \item \textbf{Contact Rates:} The matching function is $M(u_j, v_j)$. The worker job-finding rate is $\lambda(\theta_j) = M(u_j, v_j) / u_j$. The firm vacancy-filling rate is $q(\theta_j) = M(u_j, v_j) / v_j$.
    \item \textbf{Pool Composition:} Conditional on a firm in sector $j$ meeting a worker, the probability the worker is of type $i$ is $\pi_{ij} = (\phi_{ij} u_i) / u_j$.
\end{itemize}

\section{Production and Endogenous Separation}
\begin{itemize}
    \item \textbf{Match Quality:} Upon meeting, the firm and worker draw a match-specific idiosyncratic quality shock $x \sim G(x)$.
    \item \textbf{Output:} A producing match between a sector $j$ firm and a type $i$ worker with shock $x$ yields flow output $y_{ij}(x) = p_j a_i x$.
    \item \textbf{Shocks:} New idiosyncratic shocks $x'$ arrive at Poisson rate $\delta$, drawn from the same distribution $G(\cdot)$.
    \item \textbf{Exogenous Separation:} Matches are destroyed for exogenous reasons at rate $s$.
    \item \textbf{Endogenous Separation:} A match is formed (and continued) if and only if the joint surplus is non-negative, which defines a reservation match quality $R_{ij}$. If $x < R_{ij}$, the match separates (or is not formed).
\end{itemize}

\section{Value Functions}

\subsection{Workers}
The value of unemployment for a worker of type $i$ searching in sector $j$ is $U_{ij}$:
\begin{equation}
r U_{ij} = b + \lambda(\theta_j) \int_{R_{ij}}^\infty [W_{ij}(x) - U_{ij}] dG(x)
\end{equation}

The value of employment $W_{ij}(x)$ satisfies:
\begin{equation}
r W_{ij}(x) = w_{ij}(x) + \delta \int_{R_{ij}}^\infty [W_{ij}(x') - W_{ij}(x)] dG(x') - (\delta G(R_{ij}) + s) [W_{ij}(x) - U_{ij}]
\end{equation}

\textbf{Optimal Sorting:} Workers choose the sector that maximizes their value:
\begin{equation}
U_i = \max \{ U_{iH}, U_{iL} \}
\end{equation}
\begin{itemize}
    \item If $U_{iH} > U_{iL}$, then $\phi_{iH} = 1$.
    \item If $U_{iH} < U_{iL}$, then $\phi_{iH} = 0$.
    \item If $U_{iH} = U_{iL}$, $\phi_{iH} \in [0, 1]$ (indifferent).
\end{itemize}

\subsection{Firms}
The value of a vacancy in sector $j$ is $V_j$:
\begin{equation}
r V_j = -c_j + q(\theta_j) \sum_{i=1}^3 \pi_{ij} \int_{R_{ij}}^\infty [J_{ij}(x) - V_j] dG(x)
\end{equation}
Free entry implies $V_j = 0$ for all active sectors.

The value of a filled job $J_{ij}(x)$ satisfies:
\begin{equation}
r J_{ij}(x) = p_j a_i x - w_{ij}(x) + \delta \int_{R_{ij}}^\infty [J_{ij}(x') - J_{ij}(x)] dG(x') - (\delta G(R_{ij}) + s) J_{ij}(x)
\end{equation}

\section{Surplus and Equilibrium Conditions}
Wages are determined by continuous Nash Bargaining, where workers have bargaining power $\beta$:
\begin{equation}
W_{ij}(x) - U_{ij} = \beta S_{ij}(x) \quad \text{and} \quad J_{ij}(x) - V_j = (1-\beta) S_{ij}(x)
\end{equation}

The total match surplus $S_{ij}(x) = W_{ij}(x) - U_{ij} + J_{ij}(x) - V_j$ satisfies:
\begin{equation}
(r + s + \delta) S_{ij}(x) = p_j a_i x - r U_{ij} + \delta \int_{R_{ij}}^\infty S_{ij}(x') dG(x')
\end{equation}

\subsection{Job Destruction (Reservation Quality)}
The reservation quality $R_{ij}$ is defined by $S_{ij}(R_{ij}) = 0$:
\begin{equation}
p_j a_i R_{ij} - r U_{ij} + \frac{\delta}{r + s + \delta} \int_{R_{ij}}^\infty p_j a_i (x' - R_{ij}) dG(x') = 0
\end{equation}

\subsection{Job Creation (Free Entry)}
Imposing $V_j = 0$ yields the job creation condition for sector $j$:
\begin{equation}
\frac{c_j}{q(\theta_j)} = (1-\beta) \sum_{i=1}^3 \pi_{ij} \int_{R_{ij}}^\infty S_{ij}(x) dG(x)
\end{equation}

\subsection{Steady-State Flow Equations}
To close the model, one formulates the steady-state unemployment equations for each type $i$, equating flows into and out of unemployment. Let $e_i$ be the number of employed type $i$ workers.
\begin{equation}
u_i + e_i = m_i
\end{equation}
The flow into employment must equal the flow out of employment:
\begin{equation}
\lambda(\theta_j) \phi_{ij} u_i (1 - G(R_{ij})) = e_{ij} (\delta G(R_{ij}) + s)
\end{equation}
where $e_{ij}$ is the mass of type $i$ workers employed in sector $j$.

\section{Verification of Key Propositions (Analytical Sketch)}

\textbf{Initial Sorting Assumption:} Suppose initially $p_H > p_L$ such that Type 3 workers strictly prefer H ($U_{3H} > U_{3L}$), Type 2 workers are on the margin or prefer L ($U_{2H} < U_{2L}$), and Type 1 prefer L ($U_{1H} < U_{1L}$).

\textbf{Proposition 1: Higher productivity in H attracts lower quality workers.}\\
As $p_H$ increases, the surplus $S_{iH}(x)$ increases for all $i$, causing $U_{iH}$ to rise. Eventually, $U_{2H}$ exceeds $U_{2L}$. Type 2 workers (who are lower quality than the incumbent Type 3 workers) now switch their search to Sector H. The average quality of workers in Sector H drops because it now includes the lower-ability Type 2 cohort.

\textbf{Proposition 2: Rising H-tech productivity generates wage gains outside H-tech.}\\
When Type 2 workers leave Sector L to search in Sector H, the unemployment pool in Sector L ($u_L$) shrinks. The Job Creation condition in L depends on the expected surplus of meeting a worker. Since Type 1 remains in L, the firm composition $\pi_{1L}$ shifts. Even holding vacancies fixed, a drop in $u_L$ increases market tightness $\theta_L = v_L / u_L$. A higher $\theta_L$ increases the job finding rate $\lambda(\theta_L)$ for the remaining Type 1 workers. This increases their outside option $U_{1L}$, which, via Nash bargaining ($w_{ij}(x) = \beta p_j a_i x + (1-\beta)r U_{ij}$), directly drives up the wages for the non-high-tech workers.

\textbf{Proposition 3: Entire labor market becomes tighter.}\\
In Sector H, higher $p_H$ directly increases expected firm surplus, incentivizing a massive entry of vacancies $v_H$. This pushes $\theta_H$ up. In Sector L, the outflow of Type 2 workers directly reduces the denominator $u_L$, pushing $\theta_L$ up. Consequently, job finding rates rise across both sectors, making the entire economy tighter.
